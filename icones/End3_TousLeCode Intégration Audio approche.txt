Cependant, je dois être transparent avec toi sur la réalité technique : **tu ne peux pas simplement insérer un composant audio interactif (avec un bouton Play/Pause) directement à l'intérieur d'un seul RichTextEditor.** Actuellement, ton modèle de données stocke le contenu sous forme d'une chaîne de caractères HTML unique, et tu utilises la librairie mohamedrejeb.richeditor pour la gestion du texte. @Composable
```kotlin
fun ContenuSequentiel(blocs: MutableList<BlocNote>) {

            when (bloc) {

                        placeholder = "Écrivez ici...",

        }
    }
```

### 2. Le Gestionnaire d'Enregistrement (MediaRecorder)
Crée une nouvelle classe Kotlin (par exemple EnregistreurAudio.kt). Elle va encapsuler la logique un peu lourde du MediaRecorder. Nous allons sauvegarder les fichiers directement dans le dossier privé de ton application (filesDir), pour qu'ils soient persistants avec tes notes.
```kotlin
import android.content.Context
import android.media.MediaRecorder
import android.os.Build
import java.io.File
import java.io.IOException

class EnregistreurAudio(private val context: Context) {

    private var recorder: MediaRecorder? = null
    private var cheminFichierActuel: String? = null

    // Démarre l'enregistrement et retourne le chemin du fichier créé
    fun demarrerEnregistrement(): String? {
        // Crée un nom de fichier unique basé sur l'heure
        val nomFichier = "audio_note_${System.currentTimeMillis()}.m4a"
        val fichier = File(context.filesDir, nomFichier)
        cheminFichierActuel = fichier.absolutePath

        // Initialisation compatible avec les différentes versions d'Android
        recorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            MediaRecorder(context)
        } else {
            @Suppress("DEPRECATION")
            MediaRecorder()
        }.apply {
            setAudioSource(MediaRecorder.AudioSource.MIC)
            setOutputFormat(MediaRecorder.OutputFormat.MPEG_4)
            setAudioEncoder(MediaRecorder.AudioEncoder.AAC)
            setOutputFile(cheminFichierActuel)

            try {
                prepare()
                start()
            } catch (e: IOException) {
                e.printStackTrace()
                return null
            }
        }
        return cheminFichierActuel
    }

    fun arreterEnregistrement() {
        recorder?.apply {
            try {
                stop()
                release()
            } catch (e: Exception) {
                e.printStackTrace()
            }
        }
        recorder = null
    }
}

```
### 3. Le Gestionnaire de Lecture (MediaPlayer)
Crée une autre classe (ou mets-la dans le même fichier) pour gérer la lecture des audios enregistrés.
```kotlin
import android.media.MediaPlayer
import java.io.IOException

class LecteurAudio {

    private var player: MediaPlayer? = null
    var estEnLecture = false
        private set

    fun lire(cheminFichier: String, onLectureTerminee: () -> Unit) {
        player = MediaPlayer().apply {
            try {
                setDataSource(cheminFichier)
                prepare()
                start()
                estEnLecture = true
                
                // Quand l'audio est fini, on met à jour l'état
                setOnCompletionListener {
                    estEnLecture = false
                    onLectureTerminee()
                    release()
                }
            } catch (e: IOException) {
                e.printStackTrace()
            }
        }
    }

    fun arreter() {
        player?.apply {
            if (isPlaying) {
                stop()
            }
            release()
        }
        player = null
        estEnLecture = false
    }
}

```

### 2. Le Composant pour un Bloc de Texte
Pour que chaque zone de texte garde son propre style de façon indépendante, on crée un petit composant dédié qui met à jour le modèle à chaque frappe.
```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun BlocTexteUI(
    bloc: BlocNote.Texte,
    onHtmlChange: (String) -> Unit
) {
    val state = rememberRichTextState()

    // Charge le texte initial s'il y en a (utile pour l'édition d'anciennes notes)
    LaunchedEffect(Unit) {
        if (bloc.contenuHtml.isNotBlank() && state.toHtml() != bloc.contenuHtml) {
            state.setHtml(bloc.contenuHtml)
        }
    }

    // Écoute les changements d'état et remonte l'HTML
    LaunchedEffect(state.annotatedString) {
        onHtmlChange(state.toHtml())
    }

    RichTextEditor(
        state = state,
        modifier = Modifier
            .fillMaxWidth()
            .padding(horizontal = 20.dp, vertical = 4.dp),
        placeholder = { Text("Écrivez ici...", color = Color.Gray.copy(alpha = 0.5f)) },
        colors = RichTextEditorDefaults.richTextEditorColors(
            containerColor = Color.Transparent,
            focusedIndicatorColor = Color.Transparent,
            unfocusedIndicatorColor = Color.Transparent
        )
    )
}

```
### 3. L'Écran Complet (L'Assemblage Final)
Voici comment remplacer ton RichTextNoteEditorScreen. C'est le cœur du réacteur : il gère la liste, la demande de permission et l'ajout séquentiel.
```kotlin
import android.Manifest
import androidx.activity.compose.rememberLauncherForActivityResult
import androidx.activity.result.contract.ActivityResultContracts
import androidx.compose.foundation.lazy.LazyColumn
import androidx.compose.foundation.lazy.items
import androidx.compose.ui.platform.LocalContext

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun RichTextNoteEditorScreen(
    titreInitial: String = "",
    blocsInitiaux: List<BlocNote> = listOf(BlocNote.Texte("")), // On remplace le String HTML
    titrePage: String = "Nouvelle note",
    onRetour: () -> Unit,
    onSauvegarde: (String, List<BlocNote>) -> Unit // On retourne la liste des blocs
) {
    var titre by remember { mutableStateOf(titreInitial) }
    
    // La liste qui contient notre séquence Texte -> Audio -> Texte
    val blocs = remember { mutableStateListOf<BlocNote>().apply { addAll(blocsInitiaux) } }

    // --- GESTION DE L'AUDIO ---
    val context = LocalContext.current
    val enregistreur = remember { EnregistreurAudio(context) }
    var enEnregistrement by remember { mutableStateOf(false) }

    val permissionLauncher = rememberLauncherForActivityResult(
        contract = ActivityResultContracts.RequestPermission()
    ) { estAccorde ->
        if (estAccorde) {
            val chemin = enregistreur.demarrerEnregistrement()
            if (chemin != null) {
                enEnregistrement = true
            }
        }
    }

    Scaffold(
        topBar = {
            CenterAlignedTopAppBar(
                title = { Text(titrePage, fontSize = 16.sp, color = Color.Gray) },
                navigationIcon = {
                    IconButton(onClick = onRetour) {
                        Icon(Icons.Default.Close, "Annuler")
                    }
                },
                actions = {
                    // SAUVEGARDE
                    IconButton(onClick = {
                        onSauvegarde(
                            titre.ifBlank { "Note sans titre" },
                            blocs.toList() // On envoie toute la liste
                        )
                    }) {
                        Icon(Icons.Default.Check, "Sauvegarder", tint = MaterialTheme.colorScheme.primary)
                    }
                }
            )
        },
        // Bouton Micro Flottant en bas à droite
        floatingActionButton = {
            FloatingActionButton(
                onClick = {
                    if (enEnregistrement) {
                        // 1. On arrête l'audio
                        enregistreur.arreterEnregistrement()
                        enEnregistrement = false
                        
                        // 2. On récupère le chemin du fichier (géré dans ta classe)
                        // Note : modifie EnregistreurAudio pour récupérer le dernier chemin utilisé
                        val cheminDernierFichier = enregistreur.obtenirDernierChemin() ?: return@FloatingActionButton
                        
                        // 3. On ajoute le bloc audio
                        blocs.add(BlocNote.Audio(cheminFichier = cheminDernierFichier))
                        
                        // 4. On ajoute une nouvelle zone de texte vide en dessous
                        blocs.add(BlocNote.Texte(""))
                    } else {
                        // Demande la permission et lance l'enregistrement
                        permissionLauncher.launch(Manifest.permission.RECORD_AUDIO)
                    }
                },
                containerColor = if (enEnregistrement) Color.Red else MaterialTheme.colorScheme.primary
            ) {
                Icon(
                    imageVector = if (enEnregistrement) Icons.Default.Stop else Icons.Default.Mic,
                    contentDescription = "Micro",
                    tint = Color.White
                )
            }
        }
    ) { padding ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding)
                .imePadding()
        ) {
            
            // TITRE
            TextField(
                value = titre,
                onValueChange = { titre = it },
                placeholder = { Text("Titre de la note", fontSize = 28.sp, fontWeight = FontWeight.Bold, color = Color.Gray.copy(alpha = 0.5f)) },
                textStyle = LocalTextStyle.current.copy(fontSize = 28.sp, fontWeight = FontWeight.Bold),
                singleLine = true,
                colors = TextFieldDefaults.colors(
                    focusedContainerColor = Color.Transparent, unfocusedContainerColor = Color.Transparent,
                    focusedIndicatorColor = Color.Transparent, unfocusedIndicatorColor = Color.Transparent
                ),
                modifier = Modifier.fillMaxWidth().padding(horizontal = 20.dp)
            )

            HorizontalDivider(color = Color.Gray.copy(alpha = 0.2f))
            Spacer(modifier = Modifier.height(8.dp))

            // AFFICHAGE SÉQUENTIEL DES BLOCS
            LazyColumn(
                modifier = Modifier.fillMaxSize()
            ) {
                items(blocs, key = { it.idBloc }) { bloc ->
                    when (bloc) {
                        is BlocNote.Texte -> {

                                bloc = bloc,

                item { Spacer(modifier = Modifier.height(80.dp)) }
            }
        }
    }
}
```
